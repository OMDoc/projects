1. The CASL logic language

Concepttype
Conceptquantification
HetsDataType: data FORMULA f = Quantification QUANTIFIER [VAR_DECL] (FORMULA f) [Pos] -- pos: QUANTIFIER, semi colons, dot
Conceptuniversal
HetsDataType: data QUANTIFIER = Universal
Conceptexistential
HetsDataType: data QUANTIFIER = Existential
Conceptunique-existential
HetsDataType: data QUANTIFIER = Unique_existential
Conceptconjunction
HetsDataType: data FORMULA f = Conjunction [FORMULA f] [Pos] -- pos: "/\"s
Conceptdisjunction
HetsDataType: data FORMULA f = Disjunction [FORMULA f] [Pos] -- pos: "\/"s
Conceptimplication
HetsDataType: data FORMULA f = Implication (FORMULA f) (FORMULA f) Bool [Pos] -- pos: "=>" or "if" (True -> "=>")
Conceptequivalence
HetsDataType: data FORMULA f = Equivalence (FORMULA f) (FORMULA f) [Pos] -- pos: "<=>"
Conceptnegation
HetsDataType: data FORMULA f = Negation (FORMULA f) [Pos] -- pos: not
Concepttrue
HetsDataType: data FORMULA f = True_atom [Pos] -- pos: true
Conceptfalse
HetsDataType: data FORMULA f = False_atom [Pos] -- pos: false
Conceptpredication
HetsDataType: data FORMULA f = Predication PRED_SYMB [TERM f] [Pos] -- pos: opt. "(",commas,")"
Conceptdefinedness
HetsDataType: data FORMULA f = Definedness (TERM f) [Pos] -- pos: def
Conceptexistential-equation
HetsDataType: data FORMULA f = Existl_equation (TERM f) (TERM f) [Pos] -- pos: =e=
Conceptstrong-equation
HetsDataType: data FORMULA f = Strong_equation (TERM f) (TERM f) [Pos] -- pos: =
Conceptmembership
HetsDataType: data FORMULA f = Membership (TERM f) SORT [Pos] -- pos: in
Example

forall x:real . not(x~x) iff false

universal x : ( type , real ) . equivalence ( negation ( refl ( x , x ) ) , false )
